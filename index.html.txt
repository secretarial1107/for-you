    <script>
        // Layer 1: Background Rose Field
        function initBackground() {
            for (let i = 0; i < 50; i++) {
                const rose = document.createElement('div');
                rose.className = 'rose-particle';
                rose.innerText = 'üåπ';
                rose.style.left = Math.random() * 100 + 'vw';
                rose.style.top = Math.random() * 100 + 'vh';
                rose.style.fontSize = (Math.random() * 1 + 0.5) + 'rem';
                document.body.appendChild(rose);
                animateElement(rose);
            }
        }

        function animateElement(el) {
            const x = (Math.random() - 0.5) * 300;
            const y = (Math.random() - 0.5) * 300;
            el.animate([
                { transform: 'translate(0, 0) rotate(0deg)' },
                { transform: `translate(${x}px, ${y}px) rotate(${Math.random() * 360}deg)` }
            ], {
                duration: 5000 + Math.random() * 5000,
                iterations: Infinity,
                direction: 'alternate',
                easing: 'ease-in-out'
            });
        }

        // The Rose Garden Effect: Spawn roses on click/tap
        function spawnBouquet(e) {
            // Get coordinates for both mouse and touch
            const charX = e.pageX || (e.touches ? e.touches[0].pageX : 0);
            const charY = e.pageY || (e.touches ? e.touches[0].pageY : 0);

            for (let i = 0; i < 6; i++) {
                const r = document.createElement('div');
                r.className = 'click-rose';
                r.innerText = 'üåπ';
                r.style.left = charX + 'px';
                r.style.top = charY + 'px';
                
                // Random directions for the bouquet to "explode" outward
                const travelW = (Math.random() - 0.5) * 200 + 'px';
                const travelH = (Math.random() - 0.5) * 200 + 'px';
                r.style.setProperty('--tw', travelW);
                r.style.setProperty('--th', travelH);

                document.body.appendChild(r);
                
                // Remove from DOM after animation finishes
                setTimeout(() => r.remove(), 2000);
            }
        }

        function toggleMusic() {
            const music = document.getElementById('bg-music');
            const btn = document.getElementById('music-btn');
            if (music.paused) {
                music.play().catch(() => alert("Add an MP3 link to enable music!"));
                btn.innerText = "‚è∏ Pause Music";
            } else {
                music.pause();
                btn.innerText = "üéµ Play Music";
            }
        }

        window.onload = initBackground;
    </script>
